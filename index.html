// server.js — ToolVault Backend const express = require('express'); const bcrypt = require('bcryptjs'); const cors = require('cors'); const morgan = require('morgan'); const helmet = require('helmet'); const { v4: uuidv4 } = require('uuid'); const path = require('path'); const fs = require('fs'); const db = require('./db'); const { signToken, requireAuth } = require('./auth'); const app = express(); const PORT = process.env.PORT || 3001; // ── MIDDLEWARE ────────────────────────────────────────────── app.use(helmet({ contentSecurityPolicy: false })); app.use(cors({ origin: '*' })); app.use(express.json({ limit: '2mb' })); app.use(morgan('tiny')); // Serve static frontend app.use(express.static(path.join(__dirname, '../frontend/public'))); app.use('/admin', express.static(path.join(__dirname, '../frontend/admin'))); // ── ANALYTICS HELPER ──────────────────────────────────────── function recordPageview(req) { const data = db.get(); const today = new Date().toISOString().slice(0, 10); data.pageviews.push({ id: uuidv4(), date: today, path: req.path, referrer: req.headers.referer || 'direct', ua: (req.headers['user-agent'] || '').slice(0, 100), ip: (req.headers['x-forwarded-for'] || req.ip || '').slice(0, 45) }); // Keep last 10,000 pageviews if (data.pageviews.length > 10000) data.pageviews = data.pageviews.slice(-10000); db.save(); } function recordToolEvent(toolId, type, req) { const data = db.get(); data.toolEvents.push({ id: uuidv4(), toolId, type, date: new Date().toISOString().slice(0, 10), ip: (req.headers['x-forwarded-for'] || req.ip || '').slice(0, 45) }); if (data.toolEvents.length > 50000) data.toolEvents = data.toolEvents.slice(-50000); // Update counter on tool const tool = data.tools.find(t => t.id === toolId); if (tool) { tool[type === 'click' ? 'clicks' : 'views'] = (tool[type === 'click' ? 'clicks' : 'views'] || 0) + 1; } db.save(); } // ═══════════════════════════════════════════════════════════ // PUBLIC API // ═══════════════════════════════════════════════════════════ // GET /api/tools — list approved tools app.get('/api/tools', (req, res) => { const data = db.get(); if (data.settings.maintenanceMode) return res.status(503).json({ error: 'Maintenance mode' }); let tools = data.tools.filter(t => t.approved); const { q, category, pricing, sort } = req.query; if (q) { const lq = q.toLowerCase(); tools = tools.filter(t => t.name.toLowerCase().includes(lq) || t.description.toLowerCase().includes(lq) || t.category.toLowerCase().includes(lq) || (t.tags || []).some(tag => tag.toLowerCase().includes(lq)) ); } if (category && category !== 'All') tools = tools.filter(t => t.category === category); if (pricing && pricing !== 'all') tools = tools.filter(t => t.pricing === pricing); if (sort === 'name') tools.sort((a, b) => a.name.localeCompare(b.name)); else if (sort === 'newest') tools.sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt)); else if (sort === 'free') tools.sort((a, b) => (['free','freemium','paid'].indexOf(a.pricing)) - (['free','freemium','paid'].indexOf(b.pricing))); else tools.sort((a, b) => (b.featured ? 1 : 0) - (a.featured ? 1 : 0) || b.rating - a.rating); recordPageview(req); res.json({ tools, total: tools.length, categories: [...new Set(data.tools.filter(t=>t.approved).map(t=>t.category))] }); }); // GET /api/tools/:id — single tool app.get('/api/tools/:id', (req, res) => { const data = db.get(); const tool = data.tools.find(t => t.id === req.params.id && t.approved); if (!tool) return res.status(404).json({ error: 'Tool not found' }); recordToolEvent(tool.id, 'view', req); res.json(tool); }); // POST /api/tools/:id/click — track outbound click app.post('/api/tools/:id/click', (req, res) => { const data = db.get(); const tool = data.tools.find(t => t.id === req.params.id); if (!tool) return res.status(404).json({ error: 'Not found' }); recordToolEvent(tool.id, 'click', req); res.json({ ok: true }); }); // GET /api/categories app.get('/api/categories', (req, res) => { const data = db.get(); const cats = {}; data.tools.filter(t => t.approved).forEach(t => { cats[t.category] = (cats[t.category] || 0) + 1; }); res.json(Object.entries(cats).map(([name, count]) => ({ name, count }))); }); // POST /api/submit — submit a new tool (public) app.post('/api/submit', (req, res) => { const data = db.get(); if (!data.settings.allowSubmissions) return res.status(403).json({ error: 'Submissions are currently closed.' }); const { name, category, pricing, url, description, tags, submitterName, submitterEmail, featuredInterest } = req.body; if (!name || !category || !pricing || !url || !description) { return res.status(400).json({ error: 'Missing required fields' }); } const submission = { id: uuidv4(), name: name.trim(), category, pricing, url: url.trim(), description: description.trim(), tags: (tags || '').split(',').map(t => t.trim()).filter(Boolean), submitterName: submitterName || '', submitterEmail: submitterEmail || '', featuredInterest: featuredInterest || 'no', status: 'pending', submittedAt: new Date().toISOString() }; data.submissions.push(submission); db.save(); res.status(201).json({ ok: true, message: 'Submission received! We\'ll review it within 48 hours.' }); }); // GET /api/settings/public app.get('/api/settings/public', (req, res) => { const { siteName, siteTagline, maintenanceMode, allowSubmissions } = db.get().settings; res.json({ siteName, siteTagline, maintenanceMode, allowSubmissions }); }); // ═══════════════════════════════════════════════════════════ // ADMIN AUTH // ═══════════════════════════════════════════════════════════ app.post('/api/admin/login', async (req, res) => { const { username, password } = req.body; const data = db.get(); if (username !== data.admin.username) return res.status(401).json({ error: 'Invalid credentials' }); const ok = await bcrypt.compare(password, data.admin.passwordHash); if (!ok) return res.status(401).json({ error: 'Invalid credentials' }); const token = signToken({ username, role: 'admin' }); res.json({ token, username }); }); app.post('/api/admin/change-password', requireAuth, async (req, res) => { const { currentPassword, newPassword } = req.body; if (!newPassword || newPassword.length < 8) return res.status(400).json({ error: 'Password must be at least 8 characters' }); const data = db.get(); const ok = await bcrypt.compare(currentPassword, data.admin.passwordHash); if (!ok) return res.status(401).json({ error: 'Current password incorrect' }); data.admin.passwordHash = await bcrypt.hash(newPassword, 10); db.save(); res.json({ ok: true }); }); // ═══════════════════════════════════════════════════════════ // ADMIN — ANALYTICS // ═══════════════════════════════════════════════════════════ app.get('/api/admin/analytics', requireAuth, (req, res) => { const data = db.get(); const { days = 30 } = req.query; const since = new Date(); since.setDate(since.getDate() - parseInt(days)); const sinceStr = since.toISOString().slice(0, 10); const pvFiltered = data.pageviews.filter(p => p.date >= sinceStr); const evFiltered = data.toolEvents.filter(e => e.date >= sinceStr); // Pageviews by day const pvByDay = {}; pvFiltered.forEach(p => { pvByDay[p.date] = (pvByDay[p.date] || 0) + 1; }); // Unique visitors (by IP per day) const uvByDay = {}; pvFiltered.forEach(p => { const key = `${p.date}:${p.ip}`; if (!uvByDay[key]) uvByDay[p.date] = (uvByDay[p.date] || new Set()); if (uvByDay[p.date] instanceof Set) uvByDay[p.date].add(p.ip); }); const uniqueByDay = {}; Object.entries(uvByDay).forEach(([d, s]) => { uniqueByDay[d] = s instanceof Set ? s.size : s; }); // Generate date range const dates = []; for (let i = parseInt(days) - 1; i >= 0; i--) { const d = new Date(); d.setDate(d.getDate() - i); dates.push(d.toISOString().slice(0, 10)); } const timelineData = dates.map(d => ({ date: d, pageviews: pvByDay[d] || 0, visitors: uniqueByDay[d] || 0 })); // Top tools by views & clicks const toolViewCounts = {}; const toolClickCounts = {}; evFiltered.forEach(e => { if (e.type === 'view') toolViewCounts[e.toolId] = (toolViewCounts[e.toolId] || 0) + 1; if (e.type === 'click') toolClickCounts[e.toolId] = (toolClickCounts[e.toolId] || 0) + 1; }); const topByViews = Object.entries(toolViewCounts) .sort((a, b) => b[1] - a[1]).slice(0, 10) .map(([id, views]) => { const tool = data.tools.find(t => t.id === id); return { id, name: tool?.name || id, views, clicks: toolClickCounts[id] || 0, category: tool?.category || '' }; }); // Referrers const refCounts = {}; pvFiltered.forEach(p => { let ref = 'direct'; try { if (p.referrer && p.referrer !== 'direct') { const u = new URL(p.referrer); ref = u.hostname; } } catch {} refCounts[ref] = (refCounts[ref] || 0) + 1; }); const topReferrers = Object.entries(refCounts).sort((a, b) => b[1] - a[1]).slice(0, 8).map(([source, count]) => ({ source, count })); // Category popularity const catClicks = {}; evFiltered.filter(e => e.type === 'click').forEach(e => { const tool = data.tools.find(t => t.id === e.toolId); if (tool) catClicks[tool.category] = (catClicks[tool.category] || 0) + 1; }); const topCategories = Object.entries(catClicks).sort((a, b) => b[1] - a[1]).map(([cat, clicks]) => ({ cat, clicks })); // Summary stats const totalPV = pvFiltered.length; const totalUV = new Set(pvFiltered.map(p => p.ip)).size; const totalClicks = evFiltered.filter(e => e.type === 'click').length; const pendingSubmissions = data.submissions.filter(s => s.status === 'pending').length; res.json({ summary: { pageviews: totalPV, visitors: totalUV, toolClicks: totalClicks, pendingSubmissions, totalTools: data.tools.filter(t=>t.approved).length }, timeline: timelineData, topTools: topByViews, topReferrers, topCategories }); }); // ═══════════════════════════════════════════════════════════ // ADMIN — TOOLS CRUD // ═══════════════════════════════════════════════════════════ app.get('/api/admin/tools', requireAuth, (req, res) => { const data = db.get(); const tools = [...data.tools].sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt)); res.json(tools); }); app.post('/api/admin/tools', requireAuth, (req, res) => { const data = db.get(); const now = new Date().toISOString(); const tool = { id: 't' + uuidv4().slice(0, 8), ...req.body, tags: Array.isArray(req.body.tags) ? req.body.tags : (req.body.tags || '').split(',').map(t=>t.trim()).filter(Boolean), approved: req.body.approved !== false, featured: !!req.body.featured, clicks: 0, views: 0, submittedBy: 'admin', createdAt: now, updatedAt: now }; data.tools[idx] = updated; db.save(); res.json(updated); }); app.delete('/api/admin/tools/:id', requireAuth, (req, res) => { const data = db.get(); const idx = data.tools.findIndex(t => t.id === req.params.id);
